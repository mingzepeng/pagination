var test = function(o) {
	var html = '';
	for (var i = 0; i < o.pages.length; i++) {
		var item = o.pages[i];
		if (item === o.pageNumber || item === o.config['omission']) {
			html += '<span>'+o.pages[i]+'</span>';
		} else if( item === o.config['pre']){
			html += ' <a href="#'+(o.pageNumber-1)+'">'+item+'</a>';
		} else if( item === o.config['next']){
			html += '<a href="#'+(o.pageNumber+1)+'">'+item+'</a>';
		} else {
			html += '<a href="#'+item+'">'+item+'</a>';
		}
	};
	html += '';
	return html;
}

var totalRow = 800;
var pageSize = 20;

var pageNumber = 3;
var page = new Pagination(totalRow,pageSize);
var pages = page.get(pageNumber);

document.getElementById('test1').innerHTML = test(pages);
